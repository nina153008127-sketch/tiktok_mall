const express = require("express"); const app = express(); app.use(express.json()); let users = []; // REGISTER PAGE app.get("/register-page", (req, res) => { res.send(` Register

TokMall Register









Go to Login
`); }); // LOGIN PAGE app.get("/login-page", (req, res) => { res.send(` Login

TokMall Login







Create new account
`); }); // REGISTER API app.post("/register", (req, res) => { const { email, password, code } = req.body; if(code !== "123123"){ return res.send("Invalid invite code"); } users.push({ email, password }); res.send("User registered successfully"); }); // LOGIN API app.post("/login", (req, res) => { const { email, password } = req.body; const user = users.find(u => u.email === email && u.password === password); if(!user){ return res.send("Login failed"); } res.send("Login success"); }); // DASHBOARD app.get("/dashboard", (req, res) => { res.send("

🎉 Welcome to TokMall

"); }); app.listen(3000, () => { console.log("Server running: http://localhost:3000/register-page"); });
